3   import java.util.IdentityHashMap;
4   import java.util.Map;
5   
6   public class HashCodeDemo {
7   
8   	static void log(String line) {
9   		System.out.println(line);
10  	}
11  
12  	static void report(Object obj1, Object obj2) {
13  		log("obj1.hashCode() = " + obj1.hashCode());
14  		log("obj2.hashCode() = " + obj2.hashCode());
15  		log("(obj1 == obj2) : " + (obj1 == obj2));
16  	}
17  
18  	public static void main(String[] args) {
19  
20  		int limit = 10 * 1000 * 1000;
21  
22  		Map<Integer, Object> map = new IdentityHashMap<>(limit);
23  
24  		log("start");
25  
26  		for (int count = 0; count < limit; count++) {
27  			// Object obj = new String(Integer.toString(count));
28  			Object obj = new Object();
29  			int code = System.identityHashCode(obj);
30  			if (map.containsKey(code)) {
31  				log("Collision after making " + count + " objects.");
32  				report(map.get(obj.hashCode()), obj);
33  				System.exit(0);
34  			}
35  			map.put(code, obj);
36  		}
37  
38  		log("finish");
39  
40  	}
41  
42  }
