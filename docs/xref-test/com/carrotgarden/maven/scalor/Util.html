3   import java.io.File;
4   import java.nio.file.FileVisitOption;
5   import java.nio.file.Files;
6   import java.util.Arrays;
7   import java.util.function.Function;
8   import java.util.regex.Pattern;
9   
10  public interface Util {
11  
12  	static File[] fileListByRegex(File[] rootList, String regex) {
13  		Pattern pattern = Pattern.compile(regex);
14  		File[] fileList = Arrays.stream(rootList) //
15  				.map(root -> root.toPath()) //
16  				.map(root -> {
17  					try {
18  						return Files.walk(root, FileVisitOption.FOLLOW_LINKS);
19  					} catch (Exception e) {
20  						throw new RuntimeException("File system problem", e);
21  					}
22  				}) //
23  				.flatMap(Function.identity()) //
24  				.filter(path -> Files.isRegularFile(path)) //
25  				.filter(path -> pattern.matcher(path.toAbsolutePath().toString()).matches()) //
26  				.map(path -> path.toFile()) //
27  				.toArray(File[]::new);
28  		return fileList;
29  	}
30  
31  }
