001package com.carrotgarden.maven.scalor.selfregs;
002
003import org.eclipse.core.runtime.IExtension;
004import org.eclipse.core.runtime.IExtensionRegistry;
005
006public class ExtensionContributionFragment implements ContributionFragment {
007        
008        private final String namespace;
009        private final String id;
010
011        protected ExtensionContributionFragment(String namespace, String id) {
012                this.namespace = namespace;
013                this.id = id;
014        }
015
016        public void uninstall(IExtensionRegistry registry, Object token) {
017                String fullId = namespace + "." + id;
018                IExtension extension = registry.getExtension(fullId);
019                if (extension != null) {
020                        registry.removeExtension(extension, token);
021                }
022        }
023}
