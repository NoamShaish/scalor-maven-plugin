001package com.carrotgarden.maven.scalor.selfregs;
002
003import java.util.Collection;
004import java.util.Dictionary;
005import java.util.HashMap;
006import java.util.Map;
007
008import org.eclipse.core.runtime.CoreException;
009import org.eclipse.core.runtime.IConfigurationElement;
010import org.eclipse.core.runtime.IExecutableExtension;
011import org.eclipse.core.runtime.IExecutableExtensionFactory;
012import org.eclipse.core.runtime.IStatus;
013import org.eclipse.core.runtime.Status;
014
015/**
016 * <p>
017 * DO NOT USE.
018 * </p>
019 * 
020 * <p>
021 * This extension factory is used internally by the dynamic extension injector,
022 * and it must be part of the public API so that it can be loaded by the
023 * extension registry, however it does not need to be used directly by clients.
024 * </p>
025 * 
026 * @author Neil Bartlett
027 * 
028 */
029public class DelegatingExtensionFactory implements IExecutableExtension, IExecutableExtensionFactory {
030        
031        private static final Map<String, IExecutableExtensionFactory> factoryMapping = new HashMap<String, IExecutableExtensionFactory>();
032        
033        private String factoryId = null;
034        private IConfigurationElement config;
035        private String propertyName;
036
037        /**
038         * DO NOT USE.
039         * @deprecated
040         */
041        @Deprecated
042        public DelegatingExtensionFactory() {
043        }
044
045        public Object create() throws CoreException {
046                IExecutableExtensionFactory delegate = findDelegate();
047                if (delegate == null)
048                        throw new CoreException(new Status(IStatus.ERROR, Constants.BUNDLE_ID, 0, "Could not find extension factory delegate for factory ID:" + factoryId, null));
049
050                // Pass through configuration data -- needed by many extension types e.g. Views.
051                if (delegate instanceof IExecutableExtension)
052                        ((IExecutableExtension) delegate).setInitializationData(config, propertyName, null);
053                
054                return delegate.create();
055        }
056
057        private IExecutableExtensionFactory findDelegate() {
058                synchronized (factoryMapping) {
059                        return factoryMapping.get(factoryId);
060                }
061        }
062
063        public void setInitializationData(IConfigurationElement config, String propertyName, Object data) throws CoreException {
064                this.config = config;
065                this.propertyName = propertyName;
066                
067                if (data instanceof String) {
068                        factoryId = (String) data;
069                } else if (data instanceof Dictionary) {
070                        Dictionary<?,?> dict = (Dictionary<?,?>) data;
071                        Object factoryIdObj = dict.get(Constants.PROP_FACTORY_ID);
072                        if (factoryIdObj != null) {
073                                if (factoryIdObj instanceof String)
074                                        factoryId = (String) factoryIdObj;
075                                else
076                                        throw new CoreException(new Status(IStatus.ERROR, Constants.BUNDLE_ID, 0, "Factory ID attribute has invalid type (String required)", null));
077                        }
078                }
079                
080                if (factoryId == null)
081                        throw new CoreException(new Status(IStatus.ERROR, Constants.BUNDLE_ID, 0, "No factory ID specified", null));
082        }
083
084        static void addFactoryMapping(String id, IExecutableExtensionFactory factory) {
085                synchronized (factoryMapping) {
086                        if (factoryMapping.containsKey(id))
087                                throw new IllegalArgumentException("Factory mapping already exists");
088                        factoryMapping.put(id, factory);
089                }
090        }
091
092        static void removeFactoryMappings(Collection<? extends String> ids) {
093                synchronized (factoryMapping) {
094                        for (String id : ids) {
095                                factoryMapping.remove(id);
096                        }
097                }
098        }
099}
