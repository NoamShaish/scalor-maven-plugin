001package com.carrotgarden.maven.scalor.selfregs;
002
003import java.io.ByteArrayInputStream;
004import java.io.ByteArrayOutputStream;
005import java.io.IOException;
006import java.io.InputStream;
007import java.util.LinkedList;
008import java.util.List;
009
010import org.eclipse.core.runtime.ContributorFactoryOSGi;
011import org.eclipse.core.runtime.IContributor;
012import org.eclipse.core.runtime.IExecutableExtensionFactory;
013import org.eclipse.core.runtime.IExtensionRegistry;
014import org.jdom.Document;
015import org.jdom.JDOMException;
016import org.jdom.input.SAXBuilder;
017import org.jdom.output.XMLOutputter;
018import org.osgi.framework.Bundle;
019
020public class Contribution {
021
022        private final Document document;
023        private final List<ContributionFragment> fragments;
024        private final List<String> mappedFactoryIds = new LinkedList<String>();
025        private IContributor contributor;
026
027        public static Contribution loadContribution(InputStream stream, Bundle bundle)
028                        throws ContributionProcessingException, IOException {
029
030                // Setup the output to a stream buffer
031                try {
032
033                        // Parse the input
034                        SAXBuilder builder = new SAXBuilder();
035                        Document document = builder.build(stream);
036
037                        // Inject IDs and generate contribution fragments
038                        IdInjectionDocumentModifier idInjector = new IdInjectionDocumentModifier(getBundleNamespace(bundle));
039                        idInjector.modify(document);
040
041                        return new Contribution(document, idInjector.getFragments(), bundle);
042
043                } catch (JDOMException e) {
044                        throw new ContributionProcessingException("Error parsing contribution XML.", e);
045                }
046        }
047
048        private Contribution(Document document, List<ContributionFragment> fragments, Bundle bundle) {
049                this.document = document;
050                this.fragments = fragments;
051                contributor = ContributorFactoryOSGi.createContributor(bundle);
052        }
053
054        public boolean install(IExtensionRegistry registry, String name, Object token) {
055                XMLOutputter outputter = new XMLOutputter();
056                ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
057                try {
058                        outputter.output(document, outputStream);
059
060                        return registry.addContribution( //
061                                        new ByteArrayInputStream(outputStream.toByteArray()), //
062                                        contributor, false, name, null, token);
063
064                } catch (IOException e) {
065                        // Shouldn't happen with an in-memory stream?
066                        throw new RuntimeException("Unexpected error", e);
067                }
068        }
069
070        /**
071         * @param registry
072         *            The extension registry from which to uninstall
073         * @param token
074         *            The registry token ({@code null} if Eclipse is launched with
075         *            {@code -Declipse.registry.nulltoken=true}).
076         */
077        public void uninstall(IExtensionRegistry registry, Object token) {
078                for (ContributionFragment fragment : fragments) {
079                        fragment.uninstall(registry, token);
080                }
081                DelegatingExtensionFactory.removeFactoryMappings(mappedFactoryIds);
082        }
083
084        public void setFactoryObject(String id, IExecutableExtensionFactory factory)
085                        throws ContributionProcessingException {
086                FactoryInjector injector = new FactoryInjector();
087                injector.modify(document);
088
089                try {
090                        DelegatingExtensionFactory.addFactoryMapping(id, factory);
091                        mappedFactoryIds.add(id);
092                } catch (IllegalArgumentException e) {
093                        throw new ContributionProcessingException("Unable to map factory ID to supplied extension factory.", e);
094                }
095        }
096
097        private static String getBundleNamespace(Bundle bundle) {
098                String name = bundle.getSymbolicName();
099
100                int semiColonIndex = name.indexOf(';');
101                if (semiColonIndex > -1) {
102                        name = name.substring(0, semiColonIndex);
103                }
104
105                return name;
106        }
107
108}
