001package com.carrotgarden.maven.scalor.selfregs;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006import java.util.concurrent.atomic.AtomicLong;
007
008import org.jdom.Attribute;
009import org.jdom.Document;
010import org.jdom.Element;
011
012public class IdInjectionDocumentModifier implements IDocumentModifier {
013
014        private final AtomicLong counter;
015        private final String namespace;
016        private final List<ContributionFragment> fragments = new ArrayList<ContributionFragment>();
017        
018        public IdInjectionDocumentModifier(String namespace) {
019                this(namespace, new AtomicLong(0));
020        }
021        
022        public IdInjectionDocumentModifier(String namespace, AtomicLong counter) {
023                this.namespace = namespace;
024                this.counter = counter;
025        }
026
027        public void modify(Document document) throws ContributionProcessingException {
028                Element pluginElem = document.getRootElement();
029                if (!Constants.ELEM_PLUGIN.equals(pluginElem.getName()))
030                        throw new ContributionProcessingException("Invalid root element in plugin contribution document: " + pluginElem.getName());
031                
032                @SuppressWarnings("unchecked")
033                List<Element> children = pluginElem.getChildren();
034                
035                for (Element element : children) {
036                        String elemName = element.getName();
037                        if (Constants.ELEM_EXTENSION.equals(elemName) || Constants.ELEM_EXTENSION_POINT.equals(elemName)) {
038                                injectId(element);
039                        }
040                }
041        }
042
043        private void injectId(Element element) {
044                boolean extension = Constants.ELEM_EXTENSION.equals(element.getName());
045                String generatedId = null;
046                
047                Attribute idAttrib = element.getAttribute(Constants.ATTR_ID);
048                if (idAttrib == null) {
049                        generatedId = generateIdString();
050                        idAttrib = new Attribute(Constants.ATTR_ID, generatedId, Attribute.CDATA_TYPE);
051                        element.setAttribute(idAttrib);
052                } else {
053                        String id = idAttrib.getValue();
054                        if (id == null || id.length() == 0) {
055                                generatedId = generateIdString();
056                                idAttrib.setValue(generatedId);
057                        }
058                }
059                
060                if (generatedId != null) {
061                        if (extension) {
062                                fragments.add(new ExtensionContributionFragment(namespace, generatedId));
063                        } else {
064                                fragments.add(new ExtPointContributionFragment(namespace, generatedId));
065                        }
066                }
067        }
068
069        private String generateIdString() {
070                return Long.toString(counter.getAndIncrement());
071        }
072
073        public List<ContributionFragment> getFragments() {
074                return Collections.unmodifiableList(fragments);
075        }
076
077}
