001package com.carrotgarden.maven.scalor;
002
003public class DemoCast {
004
005        public static void main(String[] args) {
006                DemoCast instance = new DemoCast();
007                instance.doit();
008        }
009        
010        static void log(String line) {
011                System.out.println(line);
012        }
013
014        public void doit() {
015                A instA = new A();
016                B instB = new B();
017                A[] arrayA = new A[0];
018                B[] arrayB = new B[0];
019
020                log("---");
021                log("instB instanceof A : " + (instB instanceof A));
022                log("instA instanceof B : " + (instA instanceof B));
023                
024                log("---");
025                log("A.class.isInstance(instA) : " + A.class.isInstance(instA));
026                log("A.class.isInstance(instB) : " + A.class.isInstance(instB));
027                log("B.class.isInstance(instA) : " + B.class.isInstance(instA));
028                log("B.class.isInstance(instB) : " + B.class.isInstance(instB));
029                
030                log("---");
031                log("A.class.isAssignableFrom(B.class) : " + A.class.isAssignableFrom(B.class));
032                log("B.class.isAssignableFrom(A.class) : " + B.class.isAssignableFrom(A.class));
033                
034                log("---");
035                log("A.class.isInstance(arrayA) : " + A.class.isInstance(arrayA));
036                log("A.class.isInstance(arrayB) : " + A.class.isInstance(arrayB));
037                log("B.class.isInstance(arrayA) : " + B.class.isInstance(arrayA));
038                log("B.class.isInstance(arrayB) : " + B.class.isInstance(arrayB));
039                
040                log("---");
041                log("arrayA.getClass().isInstance(arrayB) : " + arrayA.getClass().isInstance(arrayB));
042                log("arrayA.getClass().isAssignableFrom(arrayB.getClass()) : " + arrayA.getClass().isAssignableFrom(arrayB.getClass()));
043                log("arrayB.getClass().isInstance(arrayA) : " + arrayB.getClass().isInstance(arrayA));
044                log("arrayB.getClass().isAssignableFrom(arrayA.getClass()) : " + arrayB.getClass().isAssignableFrom(arrayA.getClass()));
045        }
046
047        class A {
048        }
049
050        class B extends A {
051        }
052
053}
