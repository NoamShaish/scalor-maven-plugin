001package com.carrotgarden.maven.scalor;
002
003import java.util.IdentityHashMap;
004import java.util.Map;
005
006public class HashCodeDemo {
007
008        static void log(String line) {
009                System.out.println(line);
010        }
011
012        static void report(Object obj1, Object obj2) {
013                log("obj1.hashCode() = " + obj1.hashCode());
014                log("obj2.hashCode() = " + obj2.hashCode());
015                log("(obj1 == obj2) : " + (obj1 == obj2));
016        }
017
018        public static void main(String[] args) {
019
020                int limit = 10 * 1000 * 1000;
021
022                Map<Integer, Object> map = new IdentityHashMap<>(limit);
023
024                log("start");
025
026                for (int count = 0; count < limit; count++) {
027                        // Object obj = new String(Integer.toString(count));
028                        Object obj = new Object();
029                        int code = System.identityHashCode(obj);
030                        if (map.containsKey(code)) {
031                                log("Collision after making " + count + " objects.");
032                                report(map.get(obj.hashCode()), obj);
033                                System.exit(0);
034                        }
035                        map.put(code, obj);
036                }
037
038                log("finish");
039
040        }
041
042}
