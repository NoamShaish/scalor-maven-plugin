001package com.carrotgarden.maven.scalor;
002
003import java.io.File;
004import java.nio.file.FileVisitOption;
005import java.nio.file.Files;
006import java.util.Arrays;
007import java.util.function.Function;
008import java.util.regex.Pattern;
009
010public interface Util {
011
012        static File[] fileListByRegex(File[] rootList, String regex) {
013                Pattern pattern = Pattern.compile(regex);
014                File[] fileList = Arrays.stream(rootList) //
015                                .map(root -> root.toPath()) //
016                                .map(root -> {
017                                        try {
018                                                return Files.walk(root, FileVisitOption.FOLLOW_LINKS);
019                                        } catch (Exception e) {
020                                                throw new RuntimeException("File system problem", e);
021                                        }
022                                }) //
023                                .flatMap(Function.identity()) //
024                                .filter(path -> Files.isRegularFile(path)) //
025                                .filter(path -> pattern.matcher(path.toAbsolutePath().toString()).matches()) //
026                                .map(path -> path.toFile()) //
027                                .toArray(File[]::new);
028                return fileList;
029        }
030
031}
