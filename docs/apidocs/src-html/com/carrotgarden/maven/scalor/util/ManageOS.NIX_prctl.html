085        // close all files upon exec, except stdin, stdout, stderr
086        static void closeFiles() {
087                int size = LIB_C.getdtablesize();
088                for (int fd = 3; fd < size; fd++) { // above std*
089                        int flags = LIB_C.fcntl(fd, NIX_C.F_GETFD);
090                        if (flags < 0) {
091                                continue; // ignore errors
092                        }
093                        LIB_C.fcntl(fd, NIX_C.F_SETFD, flags | NIX_C.FD_CLOEXEC);
094                }
095        }

117        static int invokeProcess(String[] command) {
118                String exec = command[0];
119                StringArray args = new StringArray(command);
120                closeFiles();
121                int result = LIB_C.execv(exec, args);
